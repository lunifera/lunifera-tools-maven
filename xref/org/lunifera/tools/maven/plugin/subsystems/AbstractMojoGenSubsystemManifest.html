145 	protected SubsystemManifest buildSubsystemManifest()
146 			throws MojoExecutionException {
147 		if (subsystemManifest == null){
148 			throw new MojoExecutionException("You must declare <Subsystem-Manifest> in the subsystem's POM.") ;
149 		}
150 		org.apache.aries.subsystem.core.archive.SubsystemManifest.Builder subsystemManifestBuilder = new SubsystemManifest.Builder();
151 		String subsystemType;
152 		subsystemManifestBuilder.header(new SubsystemManifestVersionHeader());
153 
154 		if (getSubsystemType() == null) {
155 			subsystemType = "osgi.subsystem.application";
156 		} else {
157 			subsystemType = getSubsystemType().getValue();
158 		}
159 		subsystemManifestBuilder.type(calculateSubsystemTypeHeader(
160 				subsystemType, subsystemManifest.getProvisionPolicy()));
161 
162 		String importPackage = subsystemManifest.getImportPackage();
163 		if (importPackage != null && !importPackage.isEmpty())
164 			subsystemManifestBuilder.header(new ImportPackageHeader(
165 					importPackage));
166 
167 		String exportPackage = subsystemManifest.getExportPackage();
168 		if (exportPackage != null && !exportPackage.isEmpty())
169 			subsystemManifestBuilder.header(new ExportPackageHeader(
170 					exportPackage));
171 
172 		String preferredProvider = subsystemManifest.getPreferredProvider();
173 		if (preferredProvider != null && !preferredProvider.isEmpty())
174 			subsystemManifestBuilder.header(new PreferredProviderHeader(
175 					preferredProvider));
176 
177 		String provideCapability = subsystemManifest.getProvideCapability();
178 		if (provideCapability != null && !provideCapability.isEmpty())
179 			subsystemManifestBuilder.header(new ProvideCapabilityHeader(
180 					provideCapability));
181 
182 		String requireBundle = subsystemManifest.getRequireBundle();
183 		if (requireBundle != null && !requireBundle.isEmpty())
184 			subsystemManifestBuilder.header(new RequireBundleHeader(
185 					requireBundle));
186 
187 		String requireCapability = subsystemManifest.getRequireCapability();
188 		if (requireCapability != null && !requireCapability.isEmpty())
189 			subsystemManifestBuilder.header(new RequireCapabilityHeader(
190 					requireCapability));
191 
192 		String subsystemExportService = subsystemManifest
193 				.getSubsystemExportService();
194 		if (subsystemExportService != null && !subsystemExportService.isEmpty())
195 			subsystemManifestBuilder.header(new SubsystemExportServiceHeader(
196 					subsystemExportService));
197 
198 		String subsystemImportService = subsystemManifest
199 				.getSubsystemImportService();
200 		if (subsystemImportService != null && !subsystemImportService.isEmpty())
201 			subsystemManifestBuilder.header(new SubsystemImportServiceHeader(
202 					subsystemImportService));
203 
204 		String subsystemDescription = subsystemManifest
205 				.getSubsystemDescription();
206 		if (subsystemDescription != null && !subsystemDescription.isEmpty())
207 			subsystemManifestBuilder.header(new GenericHeader(
208 					"Subsystem-Description", subsystemDescription));
209 
210 		String version = subsystemManifest.getSubsystemVersion();
211 		if (version != null && !version.isEmpty()) {
212 			subsystemManifestBuilder.version(version);
213 		} else {
214 			subsystemManifestBuilder.version(calculateVersionFromProject());
215 		}
216 
217 		String symbolicName = subsystemManifest.getSubsystemSymbolicName();
218 		if (symbolicName != null && !symbolicName.isEmpty()) {
219 			subsystemManifestBuilder.symbolicName(symbolicName);
220 		} else {
221 			subsystemManifestBuilder
222 					.symbolicName(calculateSymbolicNameFromProject());
223 		}
224 
225 		String subsystemContent = subsystemManifest.getSubsystemContent();
226 		if (deriveContentsFromDependencies) {
227 			SubsystemContentHeader contentHeader = calculateContentFromProjectDependencies(subsystemContent);
228 			subsystemManifestBuilder.content(contentHeader);
229 		} else {
230 			if (subsystemContent != null && !subsystemContent.isEmpty()) {
231 				subsystemManifestBuilder.header(new SubsystemContentHeader(
232 						subsystemContent));
233 			}
234 		}
235 		return subsystemManifestBuilder.build();
236 	}

238 	/**
239 	 * This method will filter the dependencies defined in the project pom and
240 	 * select those that has scope Runtime or Compile and types
241 	 * Subsystem-Application, Subsystem-Feature, Subsystem-Composite or Bundle.
242 	 * By default all content will be mandatory unless the artifact is set as
243 	 * optional.
244 	 * 
245 	 * resolution – (mandatory| optional)
246 	 * 
247 	 * @param subsystemContentHeader
248 	 * 
249 	 * @return the Subsystem-Content header
250 	 * @throws MojoExecutionException
251 	 */
252 	private SubsystemContentHeader calculateContentFromProjectDependencies(
253 			String subsystemContentConfig) throws MojoExecutionException {
254 		SubsystemContentHeader subsystemContentHeader = null;
255 		Set<Artifact> artifacts = getProject().getDependencyArtifacts();
256 		Set<Clause> clauses = new HashSet<Clause>();
257 
258 		Map<String, MutableClause> clausesMap = extractClausesFromString(subsystemContentConfig);
259 
260 		for (Artifact artifact : artifacts) {
261 
262 			getLog().debug("Resolving dependency artifact " + artifact);
263 			ArtifactResolutionRequest request = new ArtifactResolutionRequest()
264 					.setArtifact(artifact)
265 					.setRemoteRepositories(remoteRepositories)
266 					.setLocalRepository(localArtifactRepository);
267 
268 			ArtifactResolutionResult resolutionResult = repositorySystem
269 					.resolve(request);
270 			if (resolutionResult.hasExceptions()) {
271 				throw new MojoExecutionException("Could not resolve artifact: "
272 						+ artifact, resolutionResult.getExceptions().get(0));
273 			}
274 			artifact = (Artifact) resolutionResult.getArtifacts().iterator()
275 					.next();
276 
277 			String type = artifact.getType();
278 			String scope = artifact.getScope();
279 			if (scope.equalsIgnoreCase("Runtime")
280 					|| scope.equalsIgnoreCase("Compile")) {
281 				MutableClause clause = clausesMap.get(artifact.getArtifactId());
282 				if (clause == null) {
283 					clause = new MutableClause(artifact.getArtifactId());
284 				}
285 
286 				if (type.equalsIgnoreCase("jar")
287 						|| type.equalsIgnoreCase("bundle")) {
288 					clause.setAttribute("type", "osgi.bundle");
289 				} else if (type.equalsIgnoreCase("subsystem-feature")) {
290 					clause.setAttribute("type", "osgi.subsystem.feature");
291 
292 				} else if (type.equalsIgnoreCase("subsystem-composite")) {
293 					clause.setAttribute("type", "osgi.subsystem.composite");
294 				} else if (type.equalsIgnoreCase("subsystem-application")) {
295 					clause.setAttribute("type", "osgi.subsystem.application");
296 				}
297 
298 				if (artifact.isOptional()) {
299 					clause.setDirective(Clause.DIRECTIVE_RESOLUTION, "optional");
300 				} else if (embedContents == true) {
301 					try {
302 						if (!dependenciesDirectory.exists()) {
303 							dependenciesDirectory.mkdirs();
304 						}
305 						if (artifact.getFile().isFile()) {
306 							// java7 new stuff
307 							Files.copy(artifact.getFile().toPath(), new File(
308 									dependenciesDirectory, artifact.getFile()
309 											.getName()).toPath());
310 						}
311 					} catch (IOException e) {
312 						e.printStackTrace();
313 						continue;
314 					}
315 				}
316 				clause.setAttribute(Clause.ATTRIBUTE_VERSION,
317 						maven2OsgiConverter.getVersion(artifact.getVersion()));
318 				clauses.add(clause);
319 			}
320 		}
321 		if (!clauses.isEmpty()) {
322 			subsystemContentHeader = new SubsystemContentHeader(clauses);
323 		}
324 		return subsystemContentHeader;
325 	}

327 	private SubsystemTypeHeader calculateSubsystemTypeHeader(
328 			String subsystemType, ProvisionPolicy provisionPolicy) {
329 		StringBuilder clauseStr = new StringBuilder("Subsystem-Type:");
330 		clauseStr.append(subsystemType);
331 		if (provisionPolicy == null) {
332 			clauseStr.append(";provision-policy:=rejectDependencies");
333 		} else {
334 			clauseStr.append(";provision-policy:=").append(
335 					provisionPolicy.toString());
336 		}
337 		SubsystemTypeHeader subsystemTypeHeader = new SubsystemTypeHeader(
338 				new SubsystemTypeHeader.Clause(clauseStr.toString()));
339 		return subsystemTypeHeader;
340 	}

343 		SubsystemSymbolicNameHeader symbolicNameHeader = null;
344 		symbolicNameHeader = new SubsystemSymbolicNameHeader(getProject()
345 				.getArtifactId());
346 		return symbolicNameHeader;

349 	private SubsystemVersionHeader calculateVersionFromProject() {
350 		SubsystemVersionHeader versionHeader = null;
351 		versionHeader = new SubsystemVersionHeader(
352 				maven2OsgiConverter.getVersion(getProject().getVersion()));
353 		return versionHeader;
354 	}

356 	@Override
357 	public void execute() throws MojoExecutionException, MojoFailureException {
358 		super.execute();
359 
360 		File source = getProject().getFile();
361 		if (!getBuildContext().hasDelta(source)) {
362 			return;
363 		}
364 
365 		SubsystemManifest subsystemManifest = buildSubsystemManifest();
366 
367 		if (subsystemManifest != null) {
368 			writeSubsystemManifestFile(subsystemManifest);
369 		} else {
370 			throw new MojoExecutionException(
371 					"Error occurred when building manifest file.");
372 		}
373 	}

375 	private Map<String, MutableClause> extractClausesFromString(
376 			String subsystemContentConfig) {
377 		Map<String, MutableClause> map = new HashMap<>();
378 
379 		if (subsystemContentConfig != null && !subsystemContentConfig.isEmpty()) {
380 			String[] clauses = subsystemContentConfig.split(",");
381 			for (int i = 0; i < clauses.length; i++) {
382 				MutableClause clause = new MutableClause(clauses[i]);
383 				if (clause != null) {
384 					String id = clause.getSymbolicName();
385 					map.put(id, clause);
386 				}
387 			}
388 		}
389 		return map;
390 	}

400 	synchronized void writeSubsystemManifestFile(
401 			SubsystemManifest subsystemManifest) throws MojoExecutionException,
402 			MojoFailureException {
403 		File file = subsystemManifestFile;
404 		OutputStream fos = null;
405 		try {
406 			if (!file.getParentFile().exists()) {
407 				file.getParentFile().mkdirs();
408 			}
409 			// TODO: should always override the file ?
410 			if (file.exists()) {
411 				file.delete();
412 			}
413 			fos = getBuildContext().newFileOutputStream(file);
414 			subsystemManifest.write(fos);
415 		} catch (FileNotFoundException e) {
416 			throw new MojoExecutionException("The file '"
417 					+ subsystemManifestFile + "' was not found.", e);
418 		} catch (IOException e) {
419 			throw new MojoFailureException(
420 					"A failure occurred while trying to open the file '"
421 							+ subsystemManifestFile + "'.", e);
422 		} finally {
423 			if (fos != null)
424 				IOUtils.close(fos);
425 		}
426 	}
